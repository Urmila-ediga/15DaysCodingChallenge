class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n=nums.length;
        int op[]=new int[n];
        for(int i=0;i<n;i++)
        {
            op[i]=1;
        }
        int l=1;
        int r=1;
        for(int i=0;i<n;i++)
        {
            op[i]*=l;
            l*=nums[i];
        }
        for(int i=n-1;i>=0;i--)
        {
            op[i]*=r;
            r*=nums[i];
        }
        return op;


        
    }
}
