class Solution {
    public int maxProfit(int[] prices) {
        int b=prices[0];
        int pf=0;
        for(int i=1;i<prices.length;i++)
        {
            if (prices[i]<b)
            {
                b=prices[i];
            }
            pf=Math.max(pf,prices[i]-b);
        }
        return pf;



        
    }
}
